    45→  const [recentGates, setRecentGates] = useState<Gate[]>([])

    49→
    50→  // Delegation form

    93→      method: 'POST',
    94→      headers: { 'Content-Type': 'application/json' },
    95→      body: JSON.stringify({
    96→        action: 'grant_delegation',

    98→      }),
    99→    })
   100→    const d = await r.json()
   101→    if (d.ok) { toast.success(`Delega concessa a ${delAgent}`); refresh() }
   102→    else toast.error(d.error)
   103→  }

   106→    const r = await fetch('/api/retainer', {
   107→      method: 'POST',
   108→      headers: { 'Content-Type': 'application/json' },

   119→      method: 'POST',
   120→      headers: { 'Content-Type': 'application/json' },
   121→      body: JSON.stringify({
   122→        action: 'request_approval',
   123→        agentId: gateAgent, gateAction, payload, reason: gateReason,
   124→      }),
   125→    })
   126→    const d = await r.json()

   132→      method: 'POST',

   138→      toast.success(decision === 'approved' ? 'Gate approvato' : 'Gate rifiutato')

   142→

   145→      method: 'POST',
   146→      headers: { 'Content-Type': 'application/json' },
   147→      body: JSON.stringify({
   148→        action: 'resolve_normative',
   149→        conflict: {
   150→          userInstruction: nUserInstr,
   151→          userLevel: nUserLevel,
   152→          systemPolicy: nSystemPolicy,
   153→          systemLevel: nSystemLevel,
   154→        },

   159→      setLastResolution(d)
   160→      if (d.verdict === 'block') toast.error(`BLOCK: gerarchia ${nSystemLevel} > ${nUserLevel}`)

   181→      <Tabs defaultValue="delegation" className="w-full">

   231→                ) : (

   236→                          <Badge variant="outline" className="font-mono text-[10px]">{d.agentId}</Badge>

   239→                            ? <Badge variant="secondary" className="text-[10px] bg-emerald-500">active</Badge>

   273→            <CardContent className="space-y-3">
   274→              <div className="grid md:grid-cols-2 gap-3">

   304→                <CardDescription>Richiedono decisione umana</CardDescription>
   305→              </CardHeader>
   306→              <CardContent>
   307→                <ul className="space-y-2">

   309→                    <li key={g.id} className="text-xs border border-amber-500/40 bg-amber-50 dark:bg-amber-950/20 rounded-md p-2.5">

   318→                          <CheckCircle2 className="size-3 mr-1" /> Approva
   319→                        </Button>

   321→                          <XCircle className="size-3 mr-1" /> Rifiuta

   323→                      </div>

   326→                </ul>
   327→              </CardContent>

   331→          <Card>
   332→            <CardHeader>

   334→              <CardDescription>{recentGates.length} gates totali</CardDescription>

   336→            <CardContent>
   337→              <ScrollArea className="h-48 pr-2">

   340→                    <li key={g.id} className="text-xs flex items-center gap-2 p-2 rounded-md hover:bg-muted/50 border">

   342→                      {g.status === 'rejected' && <XCircle className="size-3.5 text-red-500" />}
   343→                      {g.status === 'pending' && <Clock className="size-3.5 text-amber-500" />}
   344→                      <Badge variant="outline" className="text-[10px] font-mono">{g.agentId}</Badge>

   364→              </CardDescription>
   365→            </CardHeader>
   366→            <CardContent className="space-y-3">

   387→                      <SelectItem value="OPERATIONAL">OPERATIONAL (2)</SelectItem>
   388→                      <SelectItem value="AESTHETIC">AESTHETIC (3)</SelectItem>
   389→                    </SelectContent>
